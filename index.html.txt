<script>
    const API_BASE = "https://api.mail.tm";
    let token = "";
    let emailAddress = "";
    let refreshInterval;

    // 1. Initialize: Load Domains
    async function init() {
        try {
            const res = await fetch(`${API_BASE}/domains`);
            const data = await res.json();
            const select = document.getElementById('domain-select');
            select.innerHTML = ""; // Clear loading text
            
            // Populate dropdown
            if(data['hydra:member']) {
                data['hydra:member'].forEach(d => {
                    const opt = document.createElement('option');
                    opt.value = d.domain;
                    opt.text = "@" + d.domain;
                    select.appendChild(opt);
                });
            }
        } catch (error) {
            alert("Could not load domains. Please refresh or check internet.");
        }
    }
    
    // Run init on load
    window.onload = init;

    // 2. MONEY LOGIC: The "Fake" CPA Check
    function attemptCustomCreate() {
        const user = document.getElementById('username').value;
        if(!user) return alert("Please enter a name first!");

        // --- MONETIZATION TRICK ---
        // Before creating, we show a "Loading/Verification" step.
        // You can replace confirm() with a link to your CPA Locker.
        
        const wantToSupport = confirm("To create a CUSTOM address (Premium Feature), would you like to support us by viewing an ad?\n\n(Click OK to proceed)");
        
        if(wantToSupport) {
            // window.open("YOUR_CPA_LINK_HERE", "_blank"); // Uncomment this when you have a link
            createCustomEmail();
        }
        // --------------------------
    }

    // 3. Create Custom Email
    async function createCustomEmail() {
        const user = document.getElementById('username').value.toLowerCase();
        const domain = document.getElementById('domain-select').value.replace("@", ""); // Remove @ for API
        const fullEmail = `${user}@${domain}`;
        const password = "password123";

        register(fullEmail, password);
    }

    // 4. Create Random Email
    function createRandomEmail() {
        const randomStr = Math.random().toString(36).substring(7);
        const domain = document.getElementById('domain-select').value.replace("@", "");
        const fullEmail = `${randomStr}@${domain}`;
        const password = "password123";
        
        register(fullEmail, password);
    }

    // 5. Core Register & Login Logic
    async function register(email, password) {
        // Step A: Register
        try {
            await fetch(`${API_BASE}/accounts`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ address: email, password: password })
            });
        } catch (e) {
            console.log("Account likely exists, trying login...");
        }

        // Step B: Login (Get Token)
        try {
            const res = await fetch(`${API_BASE}/token`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ address: email, password: password })
            });
            
            if (!res.ok) throw new Error("Name unavailable");
            
            const data = await res.json();
            token = data.token;
            emailAddress = email;
            
            // Success: Switch UI
            showInbox();
        } catch (e) {
            alert("That name is taken or invalid. Please try another.");
        }
    }

    // 6. UI Switching
    function showInbox() {
        document.getElementById('create-form').style.display = 'none';
        document.getElementById('inbox-section').style.display = 'block';
        document.getElementById('current-email').innerText = emailAddress;
        
        // Start polling for messages
        fetchMessages();
        refreshInterval = setInterval(fetchMessages, 5000); // Check every 5s
    }

    // 7. Fetch Messages
    async function fetchMessages() {
        if(!token) return;
        
        try {
            const res = await fetch(`${API_BASE}/messages`, {
                headers: { "Authorization": `Bearer ${token}` }
            });
            const data = await res.json();
            renderMessages(data['hydra:member']);
        } catch (e) {
            console.error("Error fetching messages");
        }
    }

    // 8. Render Messages
    function renderMessages(messages) {
        const list = document.getElementById('email-list');
        list.innerHTML = ""; // Clear current list

        if (messages.length === 0) {
            list.innerHTML = `<div style="text-align:center; padding:20px; color:#aaa;">No emails yet. Auto-refreshing...</div>`;
            return;
        }

        messages.forEach(msg => {
            const item = document.createElement('div');
            item.className = 'email-item';
            item.innerHTML = `
                <div style="color:#555; font-size:0.8rem;">From: ${msg.from.address}</div>
                <div style="font-weight:bold; font-size:1rem; margin:5px 0;">${msg.subject}</div>
                <div style="color:#666;">${msg.intro}...</div>
            `;
            list.appendChild(item);
        });
    }

    // Utility: Copy to Clipboard
    function copyEmail() {
        navigator.clipboard.writeText(emailAddress);
        const btn = document.querySelector('.btn-copy');
        btn.innerText = "Copied!";
        setTimeout(() => btn.innerText = "Copy", 2000);
    }
